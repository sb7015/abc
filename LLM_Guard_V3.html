"""
This block defines the main LLMGuardEvaluator class with all methods
for evaluating prompt safety and measuring response times.
"""

class LLMGuardEvaluator:
    """
    A class to evaluate prompt safety using LLM Guard package with timing metrics.
    """
    
    def __init__(self, custom_scanners: List = None):
        """
        Initialize the evaluator with LLM Guard scanners.
        
        Args:
            custom_scanners: List of custom scanners to use. If None, uses default set.
        """
        print(" Initializing LLM Guard Evaluator...")
        if custom_scanners is None:
            self.scanners = self._get_default_scanners()
        else:
            self.scanners = custom_scanners
        
        print(f" LLM Guard Evaluator initialized with {len(self.scanners)} scanners")
    
    def _get_default_scanners(self) -> List:
        """
        Get default set of LLM Guard scanners for comprehensive safety evaluation.
        
        Returns:
            List of configured scanners
        """
        scanners = []
        
        # Try to add each scanner individually with error handling
        scanner_configs = [
            ("PromptInjection", lambda: PromptInjection(threshold=0.5)),
            ("Toxicity", lambda: Toxicity(threshold=0.7)),
            ("BanTopics", lambda: BanTopics(topics=["violence", "hate", "harassment", "illegal"], threshold=0.8)),
            ("Secrets", lambda: Secrets()),
            ("Sentiment", lambda: Sentiment(threshold=-0.5)),
            ("TokenLimit", lambda: TokenLimit(limit=4096)),
            ("BanSubstrings", lambda: BanSubstrings(substrings=[
                "ignore all previous instructions",
                "ignore previous instructions", 
                "override", "bypass", "admin access", "hack", "jailbreak"
            ], match_type="str")),
        ]
        
        # Try to add Code scanner with proper configuration
        try:
            # Use valid languages from the allowed list
            scanners.append(Code(languages=["Python", "JavaScript", "Java", "C", "C++", "PHP"]))
            print(" Code scanner added")
        except Exception as e:
            print(f" Code scanner failed: {e}")
            # Try with minimal valid languages
            try:
                scanners.append(Code(languages=["Python"]))
                print(" Code scanner added (minimal)")
            except Exception as e2:
                print(f" Code scanner completely failed: {e2}")
        
        # Try to add Language scanner
        try:
            scanners.append(Language(valid_languages=["en"]))
            print(" Language scanner added")
        except Exception as e:
            print(f" Language scanner failed: {e}")
        
        # Add other scanners
        for scanner_name, scanner_func in scanner_configs:
            try:
                scanner = scanner_func()
                scanners.append(scanner)
                print(f" {scanner_name} scanner added")
            except Exception as e:
                print(f" {scanner_name} scanner failed: {e}")
        
        if len(scanners) == 0:
            print(" No scanners could be initialized!")
            # Add a minimal scanner that should always work
            try:
                scanners.append(BanSubstrings(substrings=["hack", "bypass"], match_type="str"))
                print(" Minimal BanSubstrings scanner added as fallback")
            except:
                print(" Even minimal scanner failed")
        
        print(f" Total scanners successfully initialized: {len(scanners)}")
        return scanners
    
    def evaluate_single_prompt(self, prompt: str) -> Tuple[Dict[str, Any], float]:
        """
        Evaluate a single prompt for safety.
        
        Args:
            prompt: The prompt text to evaluate
            
        Returns:
            Tuple of (evaluation_results, response_time)
        """
        start_time = time.time()
        
        results = {
            'prompt': prompt,
            'is_safe': True,
            'overall_risk_score': 0.0,
            'scanner_results': {},
            'flagged_by': [],
            'risk_categories': []
        }
        
        try:
            for scanner in self.scanners:
                scanner_name = scanner.__class__.__name__
                
                try:
                    # Run the scanner
                    sanitized_prompt, is_valid, risk_score = scanner.scan(prompt)
                    
                    # Handle -1.0 risk scores (convert to meaningful values)
                    if risk_score == -1.0:
                        risk_score = 0.0 if is_valid else 0.8
                    
                    results['scanner_results'][scanner_name] = {
                        'is_valid': is_valid,
                        'risk_score': risk_score,
                        'sanitized_prompt': sanitized_prompt
                    }
                    
                    # Update overall safety status
                    if not is_valid:
                        results['is_safe'] = False
                        results['flagged_by'].append(scanner_name)
                        results['risk_categories'].append(self._get_risk_category(scanner_name))
                    
                    # Update overall risk score (take maximum)
                    results['overall_risk_score'] = max(results['overall_risk_score'], risk_score)
                    
                except Exception as scanner_error:
                    results['scanner_results'][scanner_name] = {
                        'error': str(scanner_error),
                        'is_valid': True,  # Default to safe if scanner fails
                        'risk_score': 0.0
                    }
        
        except Exception as e:
            results['error'] = str(e)
            results['is_safe'] = False
        
        end_time = time.time()
        response_time = end_time - start_time
        
        return results, response_time
    
    def _get_risk_category(self, scanner_name: str) -> str:
        """
        Map scanner names to risk categories.
        
        Args:
            scanner_name: Name of the scanner that flagged the prompt
            
        Returns:
            Risk category string
        """
        category_mapping = {
            'PromptInjection': 'Prompt Injection',
            'Toxicity': 'Toxic Content',
            'BanTopics': 'Banned Topics',
            'Secrets': 'Information Leakage',
            'Sentiment': 'Negative Sentiment',
            'Code': 'Code Injection',
            'Language': 'Language Issues',
            'TokenLimit': 'Token Limit Exceeded',
            'BanSubstrings': 'Banned Phrases'
        }
        return category_mapping.get(scanner_name, 'Unknown Risk')
    
    def evaluate_dataframe_prompts(self, df: pd.DataFrame, prompt_column: str) -> pd.DataFrame:
        """
        Evaluate all prompts in a DataFrame column.
        
        Args:
            df: DataFrame containing the prompts
            prompt_column: Name of the column containing prompts
            
        Returns:
            DataFrame with additional evaluation columns
        """
        print(f" Evaluating {len(df)} prompts using LLM Guard...")
        
        # Validate inputs
        if prompt_column not in df.columns:
            raise ValueError(f"Column '{prompt_column}' not found in DataFrame. Available columns: {list(df.columns)}")
        
        # Initialize result lists
        is_safe_list = []
        risk_score_list = []
        response_time_list = []
        flagged_by_list = []
        risk_categories_list = []
        
        # Process each prompt
        for idx, prompt in enumerate(df[prompt_column]):
            if idx % 10 == 0:  # Progress indicator
                print(f" Processing prompt {idx + 1}/{len(df)}")
            
            # Handle NaN or empty prompts
            if pd.isna(prompt) or prompt == "":
                prompt = "Empty prompt"
            
            results, response_time = self.evaluate_single_prompt(str(prompt))
            
            # Extract results
            is_safe_list.append(results['is_safe'])
            risk_score_list.append(results['overall_risk_score'])
            response_time_list.append(response_time)
            flagged_by_list.append(', '.join(results['flagged_by']))
            risk_categories_list.append(', '.join(results['risk_categories']))
        
        # Create result DataFrame with original data + new columns
        result_df = df.copy()
        result_df['llm_guard_is_safe'] = is_safe_list
        result_df['llm_guard_risk_score'] = risk_score_list
        result_df['evaluation_time_seconds'] = response_time_list
        result_df['flagged_by_scanners'] = flagged_by_list
        result_df['risk_categories'] = risk_categories_list
        
        print(" Evaluation completed!")
        return result_df

"""
This is the main function you'll use. It takes your DataFrame and column name,
performs evaluation, and returns results.
"""

def evaluate_prompts(df: pd.DataFrame, prompt_column: str = "PROMPT", auto_save: bool = True, filename: str = None) -> pd.DataFrame:
    """
    Main function to evaluate prompts from a DataFrame.
    
    Args:
        df: Your DataFrame containing the prompts
        prompt_column: Name of the column containing prompts (default: "PROMPT")
        auto_save: Whether to automatically save results to Excel (default: True)
        filename: Custom filename for Excel output (optional)
        
    Returns:
        DataFrame with evaluation results
    """
    print(" Starting LLM Guard Evaluation...")
    print(f" Dataset shape: {df.shape}")
    print(f" Columns: {list(df.columns)}")
    print(f" Evaluating column: '{prompt_column}'")
    
    # Validate inputs
    if df is None or df.empty:
        raise ValueError("DataFrame is empty or None")
    
    if prompt_column not in df.columns:
        raise ValueError(f"Column '{prompt_column}' not found. Available columns: {list(df.columns)}")
    
    # Initialize evaluator
    try:
        evaluator = LLMGuardEvaluator()
    except Exception as e:
        print(f" Failed to initialize evaluator: {e}")
        return df
    
    # Run evaluation
    try:
        start_time = time.time()
        results_df = evaluator.evaluate_dataframe_prompts(df, prompt_column)
        end_time = time.time()
        
        total_time = end_time - start_time
        avg_time = total_time / len(df)
        
        print(f"\n Evaluation completed successfully!")
        print(f" Total time: {total_time:.2f} seconds")
        print(f" Average time per prompt: {avg_time:.4f} seconds")
        
        # Quick summary
        safe_count = results_df['llm_guard_is_safe'].sum()
        unsafe_count = len(results_df) - safe_count
        safety_percentage = (safe_count / len(results_df)) * 100
        
        print(f" Results Summary:")
        print(f"   Safe prompts: {safe_count} ({safety_percentage:.1f}%)")
        print(f"   Unsafe prompts: {unsafe_count} ({100-safety_percentage:.1f}%)")
        
        # Auto-save to Excel if requested
        if auto_save:
            if filename is None:
                # Generate filename with timestamp
                from datetime import datetime
                timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
                filename = f"llm_guard_results_{timestamp}.xlsx"
            
            print(f"\n Auto-saving results to Excel...")
            if save_results_to_excel(results_df, filename):
                print(f" Results automatically saved to: {filename}")
            else:
                print(" Auto-save failed, but you can save manually using save_results_to_excel()")
        
        return results_df
        
    except Exception as e:
        print(f" Evaluation failed: {e}")
        return df

"""
Function to analyze results and provide detailed statistics.
"""

def analyze_results(results_df: pd.DataFrame):
    """
    Analyze evaluation results and provide comprehensive statistics.
    
    Args:
        results_df: DataFrame with evaluation results
    """
    if 'llm_guard_is_safe' not in results_df.columns:
        print(" No evaluation results found in DataFrame")
        return
    
    print("\n" + "="*60)
    print(" DETAILED RESULTS ANALYSIS")
    print("="*60)
    
    total_prompts = len(results_df)
    safe_prompts = results_df['llm_guard_is_safe'].sum()
    unsafe_prompts = total_prompts - safe_prompts
    
    print(f" Overall Statistics:")
    print(f"   Total prompts: {total_prompts}")
    print(f"   Safe prompts: {safe_prompts} ({safe_prompts/total_prompts*100:.1f}%)")
    print(f"   Unsafe prompts: {unsafe_prompts} ({unsafe_prompts/total_prompts*100:.1f}%)")
    
    # Timing analysis
    if 'evaluation_time_seconds' in results_df.columns:
        avg_time = results_df['evaluation_time_seconds'].mean()
        max_time = results_df['evaluation_time_seconds'].max()
        min_time = results_df['evaluation_time_seconds'].min()
        total_time = results_df['evaluation_time_seconds'].sum()
        
        print(f"\n Timing Analysis:")
        print(f"   Average time per prompt: {avg_time:.4f} seconds")
        print(f"   Fastest evaluation: {min_time:.4f} seconds")
        print(f"   Slowest evaluation: {max_time:.4f} seconds")
        print(f"   Total evaluation time: {total_time:.2f} seconds")
    
    # Risk score analysis
    if 'llm_guard_risk_score' in results_df.columns:
        avg_risk = results_df['llm_guard_risk_score'].mean()
        max_risk = results_df['llm_guard_risk_score'].max()
        
        print(f"\n Risk Score Analysis:")
        print(f"   Average risk score: {avg_risk:.4f}")
        print(f"   Maximum risk score: {max_risk:.4f}")
    
    # Category analysis (if Category column exists)
    if 'Category' in results_df.columns:
        print(f"\n Category-wise Analysis:")
        category_stats = results_df.groupby('Category').agg({
            'llm_guard_is_safe': ['count', 'sum'],
            'llm_guard_risk_score': 'mean' if 'llm_guard_risk_score' in results_df.columns else 'count'
        }).round(4)
        
        # Flatten column names
        if 'llm_guard_risk_score' in results_df.columns:
            category_stats.columns = ['total', 'safe_count', 'avg_risk']
        else:
            category_stats.columns = ['total', 'safe_count']
        
        category_stats['safety_percentage'] = (category_stats['safe_count'] / category_stats['total'] * 100).round(1)
        
        print(category_stats)
    
    # Most flagged categories
    if 'risk_categories' in results_df.columns:
        flagged_prompts = results_df[~results_df['llm_guard_is_safe']]
        if len(flagged_prompts) > 0:
            all_categories = []
            for cats in flagged_prompts['risk_categories']:
                if cats and cats != '':
                    all_categories.extend([cat.strip() for cat in cats.split(',')])
            
            if all_categories:
                print(f"\n Most Common Risk Categories:")
                category_counts = pd.Series(all_categories).value_counts()
                for cat, count in category_counts.head(5).items():
                    print(f"   {cat}: {count} prompts")

"""

"""

def save_results_to_excel(results_df: pd.DataFrame, filename: str = "llm_guard_evaluation_results.xlsx"):
    """
    Save results to Excel file with error handling.
    
    Args:
        results_df: DataFrame with evaluation results
        filename: Output filename
    """
    try:
        # Ensure we have the required columns
        required_eval_columns = ['llm_guard_is_safe', 'llm_guard_risk_score', 
                               'evaluation_time_seconds', 'flagged_by_scanners', 'risk_categories']
        
        missing_columns = [col for col in required_eval_columns if col not in results_df.columns]
        if missing_columns:
            print(f" Missing evaluation columns: {missing_columns}")
            print("Cannot save results - evaluation not completed")
            return False
        
        # Save to Excel
        results_df.to_excel(filename, index=False)
        print(f" Results saved to: {filename}")
        print(f" Saved {len(results_df)} rows and {len(results_df.columns)} columns")
        
        # Show column summary
        print(f" Columns saved:")
        input_cols = [col for col in results_df.columns if col not in required_eval_columns]
        eval_cols = [col for col in results_df.columns if col in required_eval_columns]
        
        print(f"   Input columns: {input_cols}")
        print(f"   Evaluation columns: {eval_cols}")
        
        return True
        
    except Exception as e:
        print(f" Error saving results: {e}")
        return False
